% q0024.tex — Fed Fights Foreign Forces [Numeric]
\question{
  id        = {0024},
  title     = {Fed Fights Foreign Forces},
  source    = {OBECON},
  year      = {2026},
  round     = {Seletiva IEO -- Phase A},
  language  = {English},
  topic     = {Macro},
  subtopic  = {Balance of Payments / Exchange Rate},
  type      = {numeric},
  statement = {%
    Consider a small open economy where $CA = 200 - 50e$ and
    $KA = 100(i - i^*)$, with $e$ the exchange rate (USD per domestic
    currency). Initially $i = 8\%$, $i^* = 4\%$. The foreign rate rises to
    $i^* = 6\%$; the central bank keeps $i = 8\%$.

    Find $e_\text{new} - e_\text{old}$ to maintain BOP equilibrium
    ($CA + KA = 0$). Round to nearest integer; if 0.5, round to even.
  },
  choices   = {},
  answer    = {-4},
  solution  = {%
    \textbf{BOP equilibrium condition:} $CA + KA = 0$, i.e.\
    $200 - 50e + 100(i - i^*) = 0$.

    \textbf{Before} ($i^* = 4\%$):
    \[
      200 - 50e + 100(8 - 4) = 0 \implies 600 = 50e \implies e_\text{old} = 12
    \]

    \textbf{After} ($i^* = 6\%$):
    \[
      200 - 50e + 100(8 - 6) = 0 \implies 400 = 50e \implies e_\text{new} = 8
    \]

    \[
      e_\text{new} - e_\text{old} = 8 - 12 = \mathbf{-4}
    \]

    Intuition: a higher foreign rate makes domestic assets relatively less
    attractive, reducing capital inflows ($KA$ falls). To restore BOP
    equilibrium, the domestic currency must \emph{depreciate} ($e$ falls,
    since $e$ is USD per domestic unit). Answer: $\mathbf{-4}$.
  },
}
