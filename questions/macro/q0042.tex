  subtopic  = {Keynesian Multiplier / Fiscal Policy},
  type      = {objective},
  statement = {%
    Keynesia has sticky prices and wages. The consumption function is
    $C = 10 + 0.8Y$. A fiscal stimulus increases investment by 10 billion.
    Find the Keynesian multiplier $m$ and the resulting change in output
    $\Delta Y$.
  },
  choices   = {%
    \item $m = 0.2$ and $\Delta Y = 2$.
    \item $m = 0.8$ and $\Delta Y = 8$.
    \item $m = 2$ and $\Delta Y = 20$.
    \item $m = 5$ and $\Delta Y = 50$.
  },
  answer    = {D},
  solution  = {%
    The Keynesian multiplier is:
    \[
      m = \frac{1}{1 - MPC} = \frac{1}{1 - 0.8} = \frac{1}{0.2} = 5.
    \]
    With $\Delta I = 10$ billion:
    \[
      \Delta Y = m \times \Delta I = 5 \times 10 = 50 \text{ billion.}
    \]
    (D) is correct.
  },
}
