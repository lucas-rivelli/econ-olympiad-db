  subtopic  = {Capital Structure / Investment Risk},
  type      = {objective},
  statement = {%
    William, a conservative investor, evaluates four investment options in
    Rivelli Corporation: ordinary shares, preferred shares, subordinated
    debentures, and stock options. Which presents the best choice for
    minimising investment risk?
  },
  choices   = {%
    \item Ordinary shares.
    \item Preferred shares.
    \item Subordinated debentures.
    \item Stock options.
  },
  answer    = {C},
  solution  = {%
    Capital structure priority from highest to lowest creditor protection:
    senior secured debt $>$ senior unsecured debt $>$ subordinated debt $>$
    preferred shares $>$ ordinary shares. Subordinated debentures offer fixed
    contractual interest payments (a legal obligation), priority over all
    equity holders in liquidation, and creditor status --- providing superior
    protection compared to any equity instrument. Stock options carry the
    highest risk. (C) is correct.
  },
}
