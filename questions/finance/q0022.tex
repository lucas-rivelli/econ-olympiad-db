\question{
  id        = {0022},
  title     = {American Dream},
  source    = {OBECON},
  year      = {2026},
  round     = {Seletiva IEO -- Phase A},
  language  = {English},
  topic     = {Finance},
  subtopic  = {Time Value of Money / Mortgage},
  type      = {objective},
  statement = {%
    A family wants to purchase a house worth \$500,000. They make a 20\% down
    payment and finance the remaining amount with a 30-year fixed-rate mortgage
    at an annual interest rate of 6\%. What is the approximate required monthly
    mortgage payment to fully repay the loan in this exact time frame?
  },
  choices   = {%
    \item \$1,200
    \item \$1,800
    \item \$2,400
    \item \$3,000
    \item \$3,600
  },
  answer    = {C},
  solution  = {%
    \textbf{Loan amount:} $PV = 500{,}000 \times (1 - 0.20) = \$400{,}000$

    \textbf{Parameters:} monthly rate $r = 6\%/12 = 0.5\% = 0.005$;
    $n = 30 \times 12 = 360$ payments.

    \textbf{Annuity formula:}
    \[
      \text{PMT} = PV \cdot \frac{r}{1-(1+r)^{-n}}
      = 400{,}000 \cdot \frac{0.005}{1 - (1.005)^{-360}}
    \]

    $(1.005)^{360} \approx 6.023$, so $(1.005)^{-360} \approx 0.1660$.

    \[
      \text{PMT} = 400{,}000 \cdot \frac{0.005}{1 - 0.1660}
      = 400{,}000 \cdot \frac{0.005}{0.8340}
      \approx 400{,}000 \times 0.005995
      \approx \mathbf{\$2{,}398}
    \]

    Closest answer: \textbf{\$2,400}. Answer: \textbf{(C)}.
  },
}
