  source    = {OBECON},
  year      = {2026},
  round     = {Seletiva IEO -- Phase C},
  language  = {English},
  topic     = {Finance},
  subtopic  = {Bond Pricing / Par Value},
  type      = {objective},
  statement = {%
    DeltaCorp is issuing 10-year bonds with a face value of \$1{,}000. The
    current market yield for bonds with similar risk and maturity is 8\% per
    annum. The company wants the bonds to trade at par value. What annual
    coupon rate must DeltaCorp set?
  },
  choices   = {%
    \item 6\% per annum.
    \item 7\% per annum.
    \item 8\% per annum.
    \item 10\% per annum.
  },
  answer    = {C},
  solution  = {%
    A bond trades at par when its coupon rate equals the market yield. At an
    8\% coupon rate, the present value of all cash flows equals the \$1{,}000
    face value. (C) is correct.
  },
}
