% q0020.tex — Don't Lose Interest (Bond Duration)
\question{
  id        = {0020},
  title     = {Don't Lose Interest},
  source    = {OBECON},
  year      = {2026},
  round     = {Seletiva IEO -- Phase A},
  language  = {English},
  topic     = {Finance},
  subtopic  = {Fixed Income / Bond Duration},
  type      = {objective},
  statement = {%
    Fernando is a risk-averse investor evaluating two 10-year government bonds
    with identical face values and yields: Bond~A (coupon-paying) and Bond~B
    (zero-coupon). Duration approximates interest rate sensitivity:
    \[
      \text{Duration} \approx -\frac{1}{P} \cdot \frac{\Delta P}{\Delta y}
    \]
    A higher duration means greater price sensitivity to rate changes.

    If Fernando wants to minimize interest rate risk, which bond should he
    choose?
  },
  choices   = {%
    \item Bond~A, even though coupon-paying bonds have higher duration than
          zero-coupon bonds of the same maturity.
    \item Bond~A, because coupon-paying bonds have lower duration than
          zero-coupon bonds of the same maturity.
    \item Bond~B, even though zero-coupon bonds have higher duration than
          coupon-paying bonds of the same maturity.
    \item Bond~B, because zero-coupon bonds have lower duration than
          coupon-paying bonds of the same maturity.
    \item Both bonds have identical interest rate risk since they have the
          same maturity.
  },
  answer    = {B},
  solution  = {%
    \textbf{Key fact:} For bonds of the same maturity, a coupon-paying bond
    has \emph{lower} Macaulay duration than a zero-coupon bond.

    \textbf{Why?} Duration is the weighted average time to receive cash flows.
    A coupon bond pays cash throughout its life (earlier coupons pull the
    weighted average down), while a zero-coupon bond pays everything at
    maturity. Therefore:
    \[
      D_\text{coupon} < D_\text{zero} = \text{maturity} = 10 \text{ years}
    \]

    Fernando wants to \emph{minimize} duration~$\Rightarrow$ choose the bond
    with \textbf{lower duration}~$\Rightarrow$ \textbf{Bond~A} (coupon-paying).

    Option (E) is wrong: same maturity $\neq$ same duration.
    Answer: \textbf{(B)}.
  },
}
