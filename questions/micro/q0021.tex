% q0021.tex — The Expected Cellphone (Adverse Selection)
\question{
  id        = {0021},
  title     = {The Expected Cellphone},
  source    = {OBECON},
  year      = {2026},
  round     = {Seletiva IEO -- Phase A},
  language  = {English},
  topic     = {Micro},
  subtopic  = {Information Economics / Adverse Selection},
  type      = {objective},
  statement = {%
    Suppose the cellphone market in Obeconland has new phones (sellers' min
    price R\$~1,000; buyers' WTP R\$~1,200) and used phones (sellers' min
    R\$~500; buyers' WTP R\$~700). Buyers cannot distinguish types before
    purchase. The proportion of new phones is $\pi$ ($0 \leq \pi \leq 1$),
    known to buyers. Trades are determined by bargaining. Mark the correct
    alternative.
  },
  choices   = {%
    \item If $\pi = 0$, the price of a cellphone would be $P^* = \text{R\$}~700$.
    \item The market failure is information asymmetry and the phenomenon is
          moral hazard.
    \item If $\pi < \frac{3}{5}$, the market only works for used cellphones,
          with $500 \leq P^* \leq 700$.
    \item If $\pi > 0$, both new and used cellphones will be traded.
    \item If buyers knew the type before purchase, only new cellphones would
          be traded.
  },
  answer    = {C},
  solution  = {%
    Buyers' expected WTP for a random phone: $\pi \cdot 1{,}200 + (1-\pi)
    \cdot 700 = 700 + 500\pi$.

    New phones sell only if buyers' WTP $\geq$ sellers' minimum for new:
    \[
      700 + 500\pi \geq 1{,}000 \implies \pi \geq \tfrac{3}{5}
    \]

    If $\pi < \frac{3}{5}$: buyers won't pay enough for new phones
    $\Rightarrow$ new phone sellers exit $\Rightarrow$ only used phones
    trade at some $P^* \in [500, 700]$. \textbf{(C) is correct.}

    Errors:
    \begin{itemize}
      \item \textbf{(A)} If $\pi=0$, all phones are used; price is between
            500 and 700 (not necessarily exactly 700).
      \item \textbf{(B)} The phenomenon is \emph{adverse selection} (not
            moral hazard, which arises \emph{after} a transaction).
      \item \textbf{(D)} Small $\pi$ prevents new phones from trading.
      \item \textbf{(E)} With full information, both types would trade
            (there are gains from trade for both).
    \end{itemize}
    Answer: \textbf{(C)}.
  },
}
