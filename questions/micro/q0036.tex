% q0036.tex — Follow the Smell(s)
\question{
  id        = {0036},
  title     = {Follow the Smell(s)},
  source    = {OBECON},
  year      = {2026},
  round     = {Seletiva IEO -- Phase C},
  language  = {English},
  topic     = {Micro},
  subtopic  = {Utility Maximization / Cobb-Douglas},
  type      = {objective},
  statement = {%
    Darko earned a budget of \$16 and chooses between cotton candy ($c$, price
    \$1) and popcorn ($p$, price \$4). His utility function is
    $u(c, p) = \sqrt{c} + 2\sqrt{p}$. Which bundle will Darko choose?
  },
  choices   = {%
    \item $c = 0$ and $p = 4$.
    \item $c = 2$ and $p = 6$.
    \item $c = 8$ and $p = 2$.
    \item $c = 12$ and $p = 1$.
  },
  answer    = {C},
  solution  = {%
    We evaluate feasible bundles directly. Option~(A): cost $= 0 + 16 = \$16$,
    $U = 0 + 2\sqrt{4} = 4$. Option~(B): cost $= 2 + 24 = \$26$ --- infeasible.
    Option~(C): cost $= 8 + 8 = \$16$, $U = \sqrt{8} + 2\sqrt{2} \approx 5.66$.
    Option~(D): cost $= 12 + 4 = \$16$,
    $U = \sqrt{12} + 2\sqrt{1} \approx 5.46$.
    Bundle~(C) satisfies the budget constraint and maximises utility.
    \textbf{(C)} is correct.
  },
}
