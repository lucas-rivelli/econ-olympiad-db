% q0027.tex — Follow the leader [Numeric]
\question{
  id        = {0027},
  title     = {Follow the Leader},
  source    = {OBECON},
  year      = {2026},
  round     = {Seletiva IEO -- Phase A},
  language  = {English},
  topic     = {Game Theory},
  subtopic  = {Industrial Organization / Stackelberg vs Cournot},
  type      = {numeric},
  statement = {%
    Two oil companies compete in a Stackelberg duopoly (rotating leadership)
    where the firm producing less becomes leader next period. Firm~A leads in
    period 1. Market demand: $Q = 360 - p$; both firms have $MC = 120$, no
    fixed costs; interest rate 20\%. No collusion; game is infinite-horizon.

    Find $(S_A + S_B) - (C_A + C_B)$ over 10 years, where $S_i$ is firm~$i$'s
    total output under the new rule and $C_i$ under Cournot.
  },
  choices   = {},
  answer    = {200},
  solution  = {%
    \textbf{Cournot equilibrium:}\\
    Profit: $\pi_i = (240 - q_i - q_j)q_i$; FOC: $240 - 2q_i - q_j = 0$.\\
    Symmetric: $q^C = 80$ each, $Q^C = 160$.\\
    Over 10 periods: $C_A = C_B = 800$, so $C_A + C_B = 1{,}600$.

    \textbf{Stackelberg equilibrium:}\\
    Follower BR: $q_F = \dfrac{240 - q_L}{2}$.\\
    Leader: $\max (240 - q_L - q_F)q_L = \max \dfrac{240 - q_L}{2} q_L$.\\
    FOC: $q_L^S = 120$; $q_F^S = 60$.

    \textbf{Leadership rotation:}\\
    Period 1: A leads ($q_A=120$, $q_B=60$). B produces less $\Rightarrow$
    B leads next.\\
    Period 2: B leads ($q_B=120$, $q_A=60$). A produces less $\Rightarrow$
    A leads next.\\
    Pattern alternates every period.

    Over 10 periods (A leads in 1,3,5,7,9; B leads in 2,4,6,8,10):
    \[
      S_A = 5 \times 120 + 5 \times 60 = 600 + 300 = 900
    \]
    \[
      S_B = 5 \times 60 + 5 \times 120 = 300 + 600 = 900
    \]
    \[
      S_A + S_B = 1{,}800
    \]

    \textbf{Difference:}
    \[
      (S_A + S_B) - (C_A + C_B) = 1{,}800 - 1{,}600 = \mathbf{200}
    \]

    The rotating-Stackelberg policy \emph{increases} total output (closer to
    competitive level), which was the government's goal to reduce market
    power externalities.
  },
}
